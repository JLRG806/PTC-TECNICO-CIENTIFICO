create table Usuarios (
id_usuario Serial primary key,
nombre_usuario varchar (80),
email_usuario varchar (50) unique,
contrasena_usuario varchar (25),
estado_usuario varchar (15) default 'Activo' check (estado_usuario in ('Activo', 'Inactivo'))
)
					 
create table Niveles (
id_nivel Serial primary key,
nivel_estudiante varchar (30)
)

create table Secciones (
id_seccion Serial primary key,
id_nivel int references Niveles(id_nivel),
seccion_estudiante varchar (4)
)

create table Especialidad (
id_especialidad Serial primary key,
especialidad_estudiante varchar (25)
)

create table Estudiantes (
id_estudiante Serial primary key,
nombre_estudiante varchar (80),
apellidos_estudiante varchar (80),
email_estudiante varchar (30) unique,
id_seccion int references Secciones(id_seccion),
id_especialidad int references Especialidad(id_especialidad)
)

create table Docentes (
id_docente Serial primary key,
nombre_docente varchar (80),
email_docente varchar (100) unique,
edad_docente date,
telefono_docente varchar(15),
dui_docente varchar(15) unique NULL
)

create table Aulas (
id_aula Serial primary key,
nombre_aula varchar (20) unique,
ubicacion_aula varchar (100),
imagen_aula varchar(255)
)

create table Grados (
id_grado Serial primary key,
id_aula int references Aulas(id_aula),
id_docente int references Docentes(id_docente),
id_seccion int references Secciones(id_seccion),
id_especialidad int references Especialidad(id_especialidad)
)

create table Evaluadores (
id_evaluador Serial primary key,
nombre_evaluador varchar (40),
apellidos_evaluador varchar(40),
email_evaluador varchar (320) unique,
telefono_evaluador varchar(10),
lugar_procedencia varchar (50),
ocupacion varchar(30),
estado_evaluador varchar (20) default 'No asignado' check (estado_evaluador in ('Convocado', 'No asignado', 'Presente', 'Evaluando', 'Nota pendiente de procesar', 'Nota procesada', 'Inactivo'))
)

create table Proyectos (
id_proyecto Serial primary key,
nombre_proyecto varchar(50),
descripcion_proyecto varchar(255),
codigo_proyecto varchar (15),
id_grado int references Grados(id_grado)
)

create table Evaluacion (
id_evaluacion Serial primary key,
id_grado int references Grados(id_grado),
id_evaluador int references Evaluadores(id_evaluador),
observaciones varchar (200)
)						

create table Detalle_Proyecto (
id_det_proyecto Serial primary key,
id_proyecto int references Proyectos(id_proyecto),
id_estudiante int references Estudiantes(id_estudiante),
puesto_estudiante varchar(20) check (puesto_estudiante in ('Coordinador', 'Sub-coordinador', 'Tesorero', 'Secretario', 'Vocero'))
)

create table Tipo_Equipo (
id_tipo_equipo Serial primary key,
tipo_equipo varchar (20) unique,
imagen_tipo_equipo varchar(255)
)

create table Equipo (
id_equipo Serial primary key,
nombre_equipo varchar (30) not null,
descripcion_equipo varchar (200),
cantidad int not null,
id_estudiante int references Estudiantes(id_estudiante),
id_tipo_equipo int references Tipo_Equipo(id_tipo_equipo),
estado_equipo varchar (30) default 'Pendiente de confirmar entrada' check (estado_equipo in ('Pendiente de confirmar entrada', 'Entrada Confirmada', 'Pendiente de confirmar salida', 'Salida Confirmada'))
)

create table Imagen_Equipo (
id_imagen_equipo Serial primary key,
id_equipo int references Equipo(id_equipo),
imagen_equipo varchar(255)
)

create table Bitacora (
id_bitacora Serial primary key,
accion varchar (100),
persona varchar (40),
fecha_hora timestamp,
tipo_bitacora varchar(30)
)
